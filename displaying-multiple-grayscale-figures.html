displaying multiple grayscale figures in python's matplotlib

Date 2014-06-06T22:12:00, Fri Tags python / ipython

from matplotlib import pyplot as plt import matplotlib.cm as cm plt.figure()  # without this it display one after another plt.imshow(image_one, cmap=cm.gray) # without cm.gray it displays grayscale images in colormap plt.imshow(image_two, cmap=cm.gray) # plt.show()

use %matplotlib inline in ipython notebook to display image inplace
