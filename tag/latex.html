crop included graphics in latex

\documentclass{article}

\usepackage{graphicx}

\begin{document}
% crop left
\includegraphics[trim={5cm 0 0 0},clip]{path-to-image}
% crop right
\includegraphics[trim={0 0 5cm 0},clip]{path-to-image}
\end{document}
